-- 10
db.clientes.aggregate([
    {
        $lookup: {
            from: "facturas",
            localField: "nro_cliente",
            foreignField: "nro_cliente",
            as: "gasto_info"
        }
    },
    {
        $unwind: {
            path: "$gasto_info",
            preserveNullAndEmptyArrays: true
        }
    },
    {
        $group: {
            _id: "$nro_cliente",
            nombre: { $first: "$nombre" },
            apellido: { $first: "$apellido" },
            gasto: { $sum: "$gasto_info.total_con_iva" }
        }
    },
    {
        $project: {
            _id: 0,
            nombre: 1,
            apellido: 1,
            gasto: { $ifNull: ["$gasto", 0] }
        }
    },
    { $count: 'total'}
])
